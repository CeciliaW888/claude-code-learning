// 终极项目：Agent CLI 主入口
import Anthropic from "@anthropic-ai/sdk";
import { Command } from "commander";

const client = new Anthropic();

// 1. CLI 入口
const program = new Command()
  .argument("[prompt]", "初始提示")
  .option("--model <model>", "模型", "claude-sonnet-4-20250514")
  .parse();

// 2. 注册工具
const tools = [bashTool, fileReadTool, fileWriteTool, fileEditTool];

// 3. 构建 System Prompt
const systemPrompt = buildSystemPrompt({ tools, claudeMd });

// 4. Agent 循环
async function run(prompt: string) {
  const messages = [{ role: "user", content: prompt }];

  while (true) {
    const res = await client.messages.create({
      model: program.opts().model,
      system: systemPrompt,
      messages,
      tools: tools.map(toAPI),
    });

    if (res.stop_reason === "end_turn") {
      console.log(getTextContent(res));
      break;
    }

    await executeTools(res, messages);
  }
}

run(program.args[0]);
