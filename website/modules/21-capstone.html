核心代码实现

最终项目将这些模块串联：入口解析参数 → 初始化状态 → 启动 REPL → 用户输入进入 Agent 循环 → 循环中调用工具 → 权限检查 → 执行 → 返回结果 → 追踪成本。

关键洞察： 一个好的 AI Agent 不在于功能多少，而在于核心循环的健壮性——输入处理、API 调用、工具执行、错误恢复的每一步都要可靠。

💻 核心代码

// 完整 Agent CLI 入口 + 循环 + REPL
import Anthropic from "@anthropic-ai/sdk";
import * as readline from "readline";

const client = new Anthropic();

// 工具定义
const tools = [
  { name: "ReadFile", /* ... */ },
  { name: "WriteFile", /* ... */ },
  { name: "Bash", /* ... */ },
];

// Agent 循环
async function agentLoop(messages: Message[]) {
  const costTracker = { totalCost: 0 };

  while (true) {
    const response = await client.messages.create({
      model: "claude-sonnet-4-20250514",
      max_tokens: 8192,
      system: buildSystemPrompt(),  // Day 11: 上下文管理
      messages,
      tools,
    });

    // Day 19: 成本追踪
    updateCost(costTracker, response.usage);

    if (response.stop_reason === "end_turn") {
      console.log(extractText(response));
      break;
    }

    // Day 5/8/9: 工具执行
    messages.push({ role: "assistant", content: response.content });
    const results = [];
    for (const block of response.content) {
      if (block.type === "tool_use") {
        // Day 15: 权限检查
        await checkPermission(block.name, block.input);
        const result = await executeTool(block);
        results.push({ type: "tool_result", tool_use_id: block.id, content: result });
      }
    }
    messages.push({ role: "user", content: results });
  }

  console.log(`\n总成本: $${costTracker.totalCost.toFixed(4)}`);
}

// REPL 交互循环
const rl = readline.createInterface({ input: process.stdin, output: process.stdout });
const messages: Message[] = [];

function prompt() {
  rl.question("You> ", async (input) => {
    if (input === "/quit") { rl.close(); return; }
    messages.push({ role: "user", content: input });
    await agentLoop(messages);
    prompt();  // 继续等待下一轮输入
  });
}

prompt();
