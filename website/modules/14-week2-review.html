💻 核心代码

// 带工具的 Agent CLI 框架
const tools: ToolDef[] = [
  {
    name: "ReadFile",
    description: "读取指定文件的内容",
    inputSchema: { type: "object", properties: {
      path: { type: "string" }
    }, required: ["path"] },
    execute: async ({ path }) => ({
      output: await fs.readFile(path, "utf-8"),
    }),
  },
  {
    name: "Bash",
    description: "执行 shell 命令（只读命令自动放行）",
    inputSchema: { type: "object", properties: {
      command: { type: "string" }
    }, required: ["command"] },
    execute: async ({ command }) => {
      if (!isReadOnly(command)) {
        const ok = await confirm(`执行: ${command}?`);
        if (!ok) return { error: "用户拒绝" };
      }
      return { output: execSync(command).toString() };
    },
  },
];

// 使用 Week 1 的 agentLoop，传入工具列表
await agentLoop(userInput, tools);
